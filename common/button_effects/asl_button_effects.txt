asl_close_options = {
	potential = {
		always = yes
	}
	allow = {
		always = yes
	}
	effect = {
		hidden_effect = {
			remove_country_flag = asl_options_opened
		}
	}
}

asl_decrease_options = {
	potential = {
		always = yes
	}
	allow = {
		check_variable = {
			which = asl_allowed_species
			value > 1
		}
	}
	effect = {
		hidden_effect = {
			every_country = {
				subtract_variable = {
					which = asl_allowed_species
					value = 1

				}
				set_country_flag = asl_option_set
			}
		}
	}
}

asl_increase_options = {
	potential = {
		always = yes
	}
	allow = {
		check_variable = {
			which = asl_allowed_species
			value < 20
		}
	}
	effect = {
		hidden_effect = {
			every_country = {
				change_variable = {
					which = asl_allowed_species
					value = 1
				}
				set_country_flag = asl_option_set
			}
		}
	}
}
